% load packages
\usepackage{amsmath,amsfonts,float,makecell,titletoc,titlesec,tocloft,lineno,booktabs,subfiles,textcomp,tabularx, etoolbox}
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\usepackage[utf8]{inputenc}
\usepackage[onehalfspacing]{setspace}
\usepackage[autostyle]{csquotes}

% format captions
\usepackage[labelfont={small,bf}, labelsep=space, font={small}]{caption}

% format section headers
\titleformat*{\section}{\large\bfseries}
\titleformat*{\subsection}{\normalsize\itshape}

% make figures static
\let\origfigure\figure
\let\endorigfigure\endfigure
\renewenvironment{figure}[1][2] {
	\expandafter\origfigure\expandafter[H]
} {
	\endorigfigure
}

% make tables in fontnote size font
\AtBeginEnvironment{multicols}{\scriptsize}{}{}
\AtBeginEnvironment{multicols}{\singlespacing}

% create new environment for table
% modified from https://tex.stackexchange.com/a/322036/174997
\newcommand{\entry}[5]{%
  \makebox[\columnwidth][s]{\num{#1}\hfil\num{#2}\hfil\num{#3}\hfil\num{#4}\hfil\num{#5}} \par
}
\newenvironment{pseudotabular}%
 {\setlength\columnsep{1cm}%
  \begin{multicols}{2}
  \footnotesize
  \setlength\parindent{0pt}}
 {\end{multicols}}
